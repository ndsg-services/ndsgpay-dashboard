Base URL
# Production NDSGPay
https://ndsgpay-backend.vercel.app/api

# Authentification : utilisez votre clé secrète NDSGPay (sk_live_)
# NE PAS utiliser un token PawaPay — NDSGPay gère PawaPay pour vous
Endpoints
POST /api/deposits Initier une collecte 
const res = await fetch("https://ndsgpay-backend.vercel.app/api/pay/deposits", {
  method: "POST",
  headers: { "Authorization": `Bearer ${votre_sk_live_xxx}` },
  body: JSON.stringify({
    depositId: uuid(),          // ID unique de votre côté
    amount: "5000",              // en XAF
    currency: "XAF",
    correspondent: "VOTRE_CORRESPONDENT",  // ex: AIRTEL_GAB, MTN_MOMO_CMR
    payer: { type: "MSISDN", address: { value: "NUMERO_CLIENT" } },  // ex: 24177000000, 237600000000
    customerTimestamp: new Date().toISOString(),
    statementDescription: "Commande #1042"
  })
});
POST /api/payouts Envoyer des fonds 
const res = await fetch("https://ndsgpay-backend.vercel.app/api/pay/payouts", {
  method: "POST",
  headers: { "Authorization": `Bearer ${votre_sk_live_xxx}` },
  body: JSON.stringify({
    payoutId: uuid(),
    amount: "10000",
    currency: "XAF",
    correspondent: "VOTRE_CORRESPONDENT",  // ex: AIRTEL_GAB, MTN_MOMO_CMR
    recipient: { type: "MSISDN", address: { value: "NUMERO_BENEFICIAIRE" } },  // ex: 24166000000, 237600000000
    customerTimestamp: new Date().toISOString(),
    statementDescription: "Remboursement client"
  })
});
POST /api/webhooks/pawapay Callback de statut 
// PawaPay envoie ce payload sur votre URL webhook

  "status": "COMPLETED",   // ou FAILED

  "correspondent": "VOTRE_CORRESPONDENT"
}
