Endpoints
POST /api/deposits Initier une collecte 
const res = await fetch("https://api.sandbox.pawapay.cloud/deposits", {
  method: "POST",
  headers: { "Authorization": `Bearer ${API_TOKEN}` },
  body: JSON.stringify({
    depositId: uuid(),          // ID unique de votre côté
    amount: "5000",              // en XAF
    currency: "XAF",
    correspondent: "AIRTEL_GA",   // ou MOOV_GA
    payer: { type: "MSISDN", address: { value: "24177000000" } },
    customerTimestamp: new Date().toISOString(),
    statementDescription: "Commande #1042"
  })
});
POST /api/payouts Envoyer des fonds 
const res = await fetch("https://api.sandbox.pawapay.cloud/payouts", {
  method: "POST",
  headers: { "Authorization": `Bearer ${API_TOKEN}` },
  body: JSON.stringify({
    payoutId: uuid(),
    amount: "10000",
    currency: "XAF",
    correspondent: "MOOV_GA",
    recipient: { type: "MSISDN", address: { value: "24166000000" } },
    customerTimestamp: new Date().toISOString(),
    statementDescription: "Remboursement client"
  })
});
